% 环

\begin{definition}{环}
一个\textbf{环}是一个集合 $R$ 与两种运算加 “$+$” 和乘 “$\vdot$” 的定义， 
\begin{itemize}
\item $(a + b) = c = a + (b + c)$
\item $a + b = b + a$
\item $a + 0 = a$
\item $a + (-a) = 0$
\item $(a\cdot b)\cdot c = a \cdot (b \cdot c)$
\item $a \cdot 1 = a$ $1 \cdot a = a$
\end{itemize}

\end{definition}
