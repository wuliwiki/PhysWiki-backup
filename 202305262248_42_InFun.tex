若 $f(x)$ 是区间 $[a,b]$ 上的连续函数，则 $f(x)$ 可积。
\end{theorem}

\begin{theorem}{有限间断点的连续函数必可积}
若 $f(x)$ 是区间 $[a,b]$ 上除在有限个点外都连续的函数，则 $f(x)$ 可积。
\end{theorem}

\begin{theorem}{单调有界函数必可积}
若 $f(x)$ 是区间 $[a,b]$ 上的单调有界函数，则 $f(x)$ 可积。
\end{theorem}
