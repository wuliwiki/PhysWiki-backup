\pentry{布洛赫理论\upref{Bloch}}

\footnote{参考\cite{GriffQ}}本文使用原子单位制\upref{AU}． 势能为
\begin{equation}
V(x) = \alpha \sum_{j\in \mathbb Z} \delta(x - ja)
\end{equation}
\begin{equation}
k = \sqrt{2mE}
\end{equation}
在 $x = ja$ 处匹配边界条件， 得
\begin{equation}
\cos(K a) = \cos(ka) + \frac{m\alpha}{k}\sin(ka)
\end{equation}
