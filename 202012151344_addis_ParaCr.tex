% 抛物线坐标系

\begin{equation}
\xi = r + z
\end{equation}
\begin{equation}
\eta = r - z
\end{equation}
\begin{equation}
\phi = \Arctan(y, x)
\end{equation}

\begin{equation}
\laplacian = \frac{4}{\xi + \eta} \qty[\pdv{\xi}\qty(\xi\pdv{\xi}) + \pdv{\eta}\qty(\eta\pdv{\eta})] + \frac{1}{\xi\eta}\pdv[2]{\phi}
\end{equation}
