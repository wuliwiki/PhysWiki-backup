\footnote{本文参考 Wikipedia \href{https://en.wikipedia.org/wiki/Hermite_polynomials}{相关页面}．}Hermite 多项式，中文也译作埃尔米特多项式．

\begin{equation}
H_n(u) \equiv (- 1)^n \E^{u^2} \dv[n]{u} \qty(\E^{-u^2})
\end{equation}
前 6 阶 Hermite 多项式分别为
\begin{equation}
\begin{array}{l}
H_0(u) = 1\\
H_1(u) = 2u\\
H_2(u) = 4u^2 - 2
\end{array}
\qquad
\begin{array}{l}
H_3(u) = 8u^3 - 12u\\
H_4(u) = 16u^4 - 48u^2 + 12\\
H_5(u) = 32u^5 - 160u^3 + 120u
\end{array}
\end{equation}
