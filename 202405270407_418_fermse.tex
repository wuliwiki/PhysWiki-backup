\pentry{线性同余\nref{nod_linmod}，同余与剩余类\nref{nod_modres}，欧拉函数（数论）\nref{nod_EulFun}}{nod_d0eb}

\begin{theorem}{费马小定理}
若 $p$ 是素数，且 $p \not{\mid}~ a$，则
\begin{equation}
a^{p-1} \equiv 1 \pmod p ~.
\end{equation}
\end{theorem}

\begin{theorem}{费马-欧拉定理}
若 $(a, m) = 1$，则
\begin{equation}
a^{\varphi(m)} \equiv 1 \pmod m ~.
\end{equation}
\end{theorem}
