\subsection{（15分）}

\[
V(x) =
\begin{cases}
\infty, & x < 0 \\
-\gamma \delta(x - a), & x > 0 \quad (\gamma, a > 0)
\end{cases}~
\]

    \[
    e^{i \lambda \sigma_z} = \cos \lambda + i \sigma_z \sin \lambda~
    \]
