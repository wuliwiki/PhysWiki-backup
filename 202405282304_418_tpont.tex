\subsection{阶}
\begin{theorem}{}
函数 $\vartheta(x)$ 与 $\psi(x)$ 的阶是 $x$：

对于足够大的 $x$（$x \ge 2$），
\begin{equation}
Ax < \vartheta(x) < Bx, ~ Cx < \psi(x) < Dx ~ ~.
\end{equation}

\end{theorem}

\subsection{证明}
我们首先引入一个引理并予以证明。
\begin{lemma}{}
\begin{equation}
\psi(x) = \vartheta(x) + \mathcal O\left( x^{1/2} (\ln x)^{1/2} \right) ~.
\end{equation}

\end{lemma}
\textbf{证明}：考虑 $\psi(x)$ 的定义，由于 $p^2 \le x$, $p^3 \le x$, $\dots$ 等价于 $p \le x^{1/2}$, $p \le x^{1/3}$, $\dots$, 故
\begin{equation}
\psi(x) = \vartheta(x) + \vartheta(x^{1/2}) + \vartheta(x^{1/3}) + \cdots = \sum_{m} \vartheta(x^{1/m}) ~.
\end{equation}
当 $x^{1/m} < 2$，也就是
\begin{equation}
m > \frac{\ln x}{\ln 2} ~~
\end{equation}
时，求和结束。由 $\vartheta(x)$ 的定义显然有 $x$ 足够大时 $\vartheta(x) < x \ln x$，故 $m \ge 2$ 时有
\begin{equation}
\vartheta(x^{1/m}) < x^{1/m} \ln x \le x^{1/2} \ln x~,
\end{equation}
且
\begin{equation}
\sum_{m \ge 2} \vartheta(x^{1/m}) = \mathcal O\left( x^{1/2} (\ln x)^{1/2} \right) ~.
\end{equation}
