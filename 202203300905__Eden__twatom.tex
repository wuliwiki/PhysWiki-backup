\begin{equation}
V(a+\delta)=V(a)+\frac{1}{2}\beta \delta^2+\text{高阶项}
\end{equation}
\textbf{忽略高阶项}，则相邻原子间作用力为
\begin{equation}
F=-\frac{\dd V}{\dd \delta}=-\beta \delta
\end{equation}
