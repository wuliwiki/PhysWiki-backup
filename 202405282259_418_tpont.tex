\subsection{阶}
\begin{theorem}{}
函数 $\vartheta(x)$ 与 $\psi(x)$ 的阶是 $x$：

对于足够大的 $x$（$x \ge 2$），
\begin{equation}
Ax < \vartheta(x) < Bx, ~ Cx < \psi(x) < Dx ~ ~.
\end{equation}

\end{theorem}

\subsection{证明}
我们首先引入一个引理并予以证明。
\begin{lemma}{}
\begin{equation}
\psi(x) = \vartheta(x) + \mathcal O\left( x^{1/2} (\ln x)^{1/2} \right) ~.
\end{equation}

\end{lemma}
