% 相对论速度变换
% 相对论|光速|洛伦兹变换

$S'$ 系相对 $S$ 系运动速度为 $u$．
\begin{equation}
v' = \frac{v + u}{1 - vu/c^2}
\end{equation}

\begin{equation}
\leftgroup{
&x' = \frac{x - vt}{\sqrt{1 - v^2/c^2}}\\
&y'= y\\
&z' = z\\
&t' = \frac{t - vx/c^2}{\sqrt{1 - v^2/c^2}}
}
\qquad
\leftgroup{
&x = \frac{x' + vt'}{\sqrt{1 - v^2/c^2}}\\
&y = y'\\
&z = z'\\
&t = \frac{t' + vx'/c^2}{\sqrt{1 - v^2/c^2}}
}
\end{equation}
