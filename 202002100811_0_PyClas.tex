% Python 的类

\begin{lstlisting}[language=python]
# 定义 person 类
class person(): 
    def __init__(self, name, age): 
        self.name = name 
        self.age = age 
    def show(self): 
        print("name is", self.name ) 
        print("age is", self.age ) 

# 生成对象
p1 = person("jason", "30") 
p2 = person("justin", "28")
# 调用成员函数
p1.show();
p2.show();
\end{lstlisting}
运行结果
\begin{lstlisting}
name is jason
age is 30
name is justin
age is 28
\end{lstlisting}
