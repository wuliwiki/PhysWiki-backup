\pentry{个体词、谓词与量词（数理逻辑）\nref{nod_prelo1}，公式的基本等价关系（数理逻辑）\nref{nod_wffEqR}}{nod_3338}
\begin{theorem}{改名规则}
\begin{equation}
(\forall x) F(x) \Leftrightarrow (\forall y) F(y) ~.
\end{equation}
\begin{equation}
(\exists x) F(x) \Leftrightarrow (\exists y) F(y) ~.
\end{equation}
\end{theorem}
\begin{theorem}{量词转换}
\begin{equation}
\neg (\forall x A(x)) \Leftrightarrow \exists x (\neg A(x)) ~.
\end{equation}
\begin{equation}
\neg (\exists x A(x)) \Leftrightarrow \forall x (\neg A(x)) ~.
\end{equation}

\end{theorem}
