\begin{theorem}{伽马函数的三种定义}
\begin{cases}  
欧拉：\displaystyle \Gamma(z):=\int^{\infty}_{0}e^{-t}t^{z-1}\text dt\\ 
高斯：\displaystyle \Gamma(z):=\lim_{n \rightarrow \infty}{\frac{n!n^z}{z(z+1)\cdots (z+n)}}=\prod_{n}^{\infty}\frac{\left( 1+\frac1n \right)^z}{\left( 1+\frac zn \right)}\\ 
魏尔斯特拉斯：\displaystyle \frac1{\Gamma(x)}:=ze^{\gamma z}\prod_{n}^{\infty}\left[ \left( 1+\frac{z}{n}\right)e^{-\frac{z}{n}}  \right] \end{cases}
\end{theorem}
