\pentry{线性映射的结构\upref{MatLS2}}

\begin{theorem}{}
线性映射 $A:X\to Y$ 中， 零值域为 $Y_1 = A(X)$， 那么任意 $b \in Y_1$ 的逆像\footnote{即所有满足 $Ax = b$ 的 $x$． 详见 “映射\upref{map}”} $X_s = A^{-1}b$ 为
\begin{equation}
X_s = x_1 + X_0
\end{equation}

任意满足 $Ax_1 = b$ 的特解 $x_1 \in X$ 加上零空间 $X_0$ 得到的集合．
\end{theorem}
