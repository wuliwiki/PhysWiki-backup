\begin{equation}
\bvec p(t) = p_0 \cos(\omega t) \uvec z
\end{equation}

\begin{equation}
V(r, \theta, t) = -\frac{p_0\omega}{4\pi\epsilon_0 c} \qty(\frac{\cos\theta}{r}) \sin[\omega(t - r/c)]
\end{equation}
\begin{equation}
\bvec A(r, \theta, t) = -\frac{\mu_0 p_0 \omega}{4\pi r} \sin[\omega(t - r/c)]\uvec z
\end{equation}
