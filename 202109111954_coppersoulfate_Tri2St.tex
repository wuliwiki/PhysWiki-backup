\subsubsection{Δ 转 Y}
\begin{equation}
R_1 = \frac{R_b R_c}{R_s} \qquad
R_2 = \frac{R_a R_c}{R_s} \qquad
R_3 = \frac{R_a R_b}{R_s}
\end{equation}
其中 $R_s = R_a + R_b + R_c$．

\subsubsection{Y 转 Δ}
\begin{equation}
R_a = \frac{R_0}{R_1} \qquad
R_b = \frac{R_0}{R_2} \qquad
R_c = \frac{R_0}{R_3}
\end{equation}
其中 $R_0 = R_1 R_2 + R_2 R_3 + R_3 R_1$．
