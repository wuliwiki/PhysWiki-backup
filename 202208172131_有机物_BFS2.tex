树与图的广度优先遍历的大致思路和 广度优先搜索（BFS）\upref{BFS}差不多，实现 BFS 需要使用一个队列来维护结点，每次先把 $1$ 号结点入队，在依次遍历每一层中的结点，并把这些结点插入到队尾，队头的元素出队，直到队列为空，就遍历完了所有元素．

\begin{lstlisting}[language=cpp]
void bfs()
{
    int hh = 0, tt = -1;
    q[ ++ tt] = 1;
    memset(d, -1, sizeof d);
    d[1] = 0;
    
    while (hh <= tt)
    {
        auto t = q[hh ++ ];
        for (int i = h[t]; ~i; i = ne[i])
        {
            int j = e[i];
            if (d[j] == -1)
            {
                d[j] = d[t] + 1;
                q[ ++ tt] = j;
            }
        }
    }
}
\end{lstlisting}
