\subsection{推导}
\begin{equation}
H^0\psi_n^2 + H^1\psi_n^1 = E_n^0\psi_n^2 + E_n^1\psi_n^1 + E_n^2\psi_n^0~.
\end{equation}
左乘任意 $\psi_m^0$ 得
\begin{equation}
\mel{\psi_m^0}{H^0}{\psi_n^2} + \mel{\psi_m^0}{H^1}{\psi_n^1} = E_n^0\braket{\psi_m^0}{\psi_n^2} + E_n^1\braket{\psi_m^0}{\psi_n^1} + \delta_{m,n}E_n^2~.
\end{equation}
