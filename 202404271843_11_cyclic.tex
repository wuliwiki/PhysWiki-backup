\begin{definition}{}
若群$G$的元素都是由某个元素生成，即$G=<a>$，则称群$G$为\textbf{循环群}（cyclic group），称$a$是循环群的生成元。
\end{definition}
\begin{example}{}
整数群$\mathbb Z$是有无限元素的循环群，群乘法为加法。
\end{example}
\begin{example}{}
模$n$同余类$\mathbb Z_n$。
\end{example}
\begin{example}{}
群$G=\{-1,-\mathrm i,1,\mathrm i\}=<\mathrm i>$。
\end{example}
