\subsubsection{3.数据操作（数据表）}
\begin{itemize}
\item 新增数据： INSERT INTO 表名 VALUES
\begin{lstlisting}[language=bash]
//向学生表中插入学生数据：张三 231119001  男 北京
INSERT INTO students
VALUES
（'张三','231119001','男','北京'）；
\end{lstlisting}

\item 查看数据：SELECT FORM 表名 
\begin{lstlisting}[language=bash]
//从学生表查询名字为张三的学号(具体查询)
SElECT ID
FROM student
WHERE name='张三'；


//从学生表查询姓名为张的学号(模糊查询)
// % 表示任意多个字符
//_ 表示任意多个任意字符

SELECT ID
FROM student
WHERE  name like '张%' ;
或者
SELECT ID
FROM student
WHERE  name like '张_' ;
\end{lstlisting}

\end{itemize}
