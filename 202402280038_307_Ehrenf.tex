\begin{theorem}{Ehrenfest 定理}
\end{theorem}
\begin{equation}
\frac{\partial\langle A\rangle}{\partial t}=\left\langle\frac{\partial A}{\partial t}\right\rangle+\frac{i}{\hbar}\langle[H, A]\rangle~.
\end{equation}
