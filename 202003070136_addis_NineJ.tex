\subsection{对称性}
任意两行行或两列交换， 在前面添加 $(-1)^S$， 其中 $S$ 是 9 个 $j$ 之和．

\subsection{特殊情况}
\begin{equation}
\Bmat{j_1 & j_2 & j_3\\ j_4 & j_5 & j_6\\ j_7 & j_8 & 0} =
\frac{\delta_{j_3, j_6}\delta_{j_7, j_8}}{\sqrt{(2j_3+1)(2j_7+1)}} (-1)^{j_2+j_3+j_4+j_7} \Bmat{j_1 & j_2 & j_3\\ j_5 & j_4 & j_7}
\end{equation}
