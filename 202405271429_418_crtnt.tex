\pentry{线性同余\nref{nod_linmod}}{nod_3099}
\begin{theorem}{中国剩余定理}
假设一组整数 $m_1, m_2, \dots, m_k$ 两两互素，则对于任意的一组整数 $a_1, a_2, \dots, a_k$，关于 $x$ 的同余方程组
\begin{equation}
\begin{aligned}
x &\equiv a_1 \pmod{m_1} \\
x &\equiv a_2 \pmod{m_2} \\
\dots   \\
~,
\end{aligned}
\end{equation}

\end{theorem}
