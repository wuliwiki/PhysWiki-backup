\begin{lstlisting}[language=python]
# 定义 person 类
class person(): # 括号可以省略
    def __init__(self, name, age): 
        self.name = name 
        self.age = age 
    def show(self): 
        print("name is", self.name ) 
        print("age is", self.age ) 

# 生成对象
p1 = person("jason", "30") 
p2 = person("justin", "28")
# 调用成员函数
p1.show();
p2.show();
\end{lstlisting}
运行结果
\begin{lstlisting}
name is jason
age is 30
name is justin
age is 28
\end{lstlisting}

来定义一个平面点类
\begin{lstlisting}[language=python]
class point: # 括号可以省略
    def __init__(self, x, y): 
        self.x = x 
        self.y = y 
    def show(self):
        print("x =", self.x, ", y =", self.y) 
    def __add__(self, other): # 定义算符 “+”
        return point(self.x + other.x, self.y + other.y)
\end{lstlisting}
