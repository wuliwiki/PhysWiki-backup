% 柯西—施瓦茨不等式

对 $\mathbb C^N$ 中有
\begin{equation}
\abs{\sum_i^N u_i^* v_i}^2 \leqslant \sum_j \abs{u_j}^2 \sum_k \abs{v_k}^2
\end{equation}

\subsection{一般情况}
\pentry{内积\upref{InerPd}}
任意内积都满足
\begin{equation}
\abs{\braket{u}{v}}^2 \leqslant \braket{u}{u} \cdot \braket{v}{v}
\end{equation}
当且仅当 $u$， $v$ 共线时取等号
