\begin{equation}
\Nabla = \uvec x \pdv{x} + \uvec y \pdv{y} + \uvec z \pdv{z}
\end{equation}

\begin{equation}
\Nabla f(x, y, z) = \uvec x \pdv{f}{x} + \uvec y \pdv{f}{y} + \uvec z \pdv{f}{z}
