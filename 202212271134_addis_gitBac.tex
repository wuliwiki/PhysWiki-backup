\begin{lstlisting}[language=bash, git-backup.sh]
repos="文件夹1 文件夹2 ..."
dir="备份目录"

for repo in $repos
do
  printf "\n\n\n===============================\n"
  echo "$repo"
  printf "===============================\n\n\n"
  if ! [ -d "$repo" ]; then
    echo "error: directory not found!"
    continue
  fi
  gitdir="${dir}${repo}.git"
  dirflag="--work-tree=\"$repo\" --git-dir=\"$gitdir\""
  echo dirflag is "$dirflag"
  echo "mkdir $gitdir"
  mkdir -p \"$gitdir\"
  echo "git init ..." &&
  git $dirflag init &&
  echo "git add -A ..." &&
  git $dirflag add -A --verbose
  # echo "git commit..." &&
  # git $dirflag commit -m 'update' &&
  # git $dirflag status'
done
\end{lstlisting}
