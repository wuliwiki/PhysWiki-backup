\subsection{基本公式}
\begin{equation}
sin^2\alpha + cos^2\alpha = 1
\end{equation}
\begin{equation}
\frac{sin\alpha}{cos\alpha} = tan\alpha
\end{equation}

\subsection{两角和与两角差}
\begin{equation}
sin(\alpha + \beta) = sin\alpha cos\beta + cos\alpha sin\beta
\end{equation}
\begin{equation}
sin(\alpha + \beta) = sin\alpha cos\beta + cos\alpha sin\beta
\end{equation}
\begin{equation}
