% 黑体辐射定律
% keys 黑体辐射|普朗克|电磁波

每个能级（$n\omega\hbar$）的平均能量
\begin{equation}
E(\nu) = \frac{h\nu}{\E^{h\nu/(k_B T) - 1}}
\end{equation}

单位体积单位频率的能量
\begin{equation}
S_\nu = \frac{8\pi h}{c^3}\frac{\nu^3}{\E^{h\nu/(k_B T) - 1}}
\end{equation}

态密度
\begin{equation}
\rho(\nu) = \frac{8\pi}{c^3}\nu^2
\end{equation}
