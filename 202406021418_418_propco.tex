\pentry{命题\nref{nod_prop}}{nod_2565}

\begin{definition}{否定}
对于一个命题 $p$， $\neg p$（或 $\sim p$）表示其否定。例如 $p$ 代表“我喜欢踢足球”，则 $\neg p$ 代表“我\textbf{不}喜欢踢足球”。
\end{definition}

\begin{definition}{合取}
对于命题 $p$ 与命题 $q$，$p \land q$ 表示这两个命题的合取，$(p \land q)$ 为真当且仅当 $p$ 与 $q$ 两者都为真。
\end{definition}
