\begin{theorem}{Bertrand 定理}
对于足够大的 $n$，那么至少存在一个素数 $p$ 使得
\begin{equation}
n < p \le 2n ~,
\end{equation}
或者等价地说，若 $p_m$ 为第 $m$ 个素数，则对于每个足够大的 $m$ 都有
\begin{equation}
p_{m+1} < 2 p_m ~.
\end{equation}

\end{theorem}
