\begin{definition}{凸集}
对于一个集合$C\subset\mathbb{R}^n,\forall\boldsymbol{x}_1,\boldsymbol{x}_2,\theta\in[0,1],$满足
$\theta\boldsymbol{x}_1+(1-\theta)\boldsymbol{x}_2\in C,$称$C$是凸集.
\end{definition}
