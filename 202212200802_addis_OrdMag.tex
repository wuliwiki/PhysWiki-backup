\begin{equation}
x
\end{equation}


\begin{itemize}
\item 小数部分的范围通常在 $1 < \abs{x} < 10$。
\item $\e{n}$ 表述小数点移动的位数。 $n > 0$ 则向右移动， $n < 0$ 则向左移动， $n=0$ 则不移动（$10^{0} = 1$）。
\item 小技巧： $x\e{-n}$ （ $1 < \abs{x} < 10$， $n > 0$）前面一共有 $n$ 个零。
\end{itemize}
